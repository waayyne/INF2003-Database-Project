How to Test All Triggers

  1. Test INSERT trigger:
    • Go to Add Product
    • Add a new product (e.g., "Test Product" with price $19.99)
    • Check the add logs table above → Should show an INSERT record

  2. Test UPDATE trigger:
    • Go to Manage Products
    • Find the product you just added and click "Edit"
    • Change something (name, price, rating, etc.)
    • Click "Update Product"
    • Check the audit logs table above → Should show an UPDATE record with specific changes

  3. Test DELETE trigger:
    • Go to Manage Products
    • Find the product you just added and click "Delete"
    • Confirm deletion
    • Check the audit logs table above → Should show a DELETE record

Trigger SQL Code

View INSERT Trigger 
CREATE TRIGGER after_product_insert
AFTER INSERT ON products
FOR EACH ROW
BEGIN
    INSERT INTO product_add_logs (
        product_id, product_name, action_type, changed_fields
    )
    VALUES (
        NEW.product_id, NEW.product_name, 'INSERT',
        CONCAT('Added new product: ', NEW.product_name, ' ($', NEW.price_usd, ')')
    );
END
View UPDATE Trigger 
CREATE TRIGGER after_product_update
AFTER UPDATE ON products
FOR EACH ROW
BEGIN
    DECLARE changes VARCHAR(2000);
    SET changes = '';
    
    -- Track product_name changes
    IF OLD.product_name != NEW.product_name THEN
        SET changes = CONCAT(changes, 'product_name: "', OLD.product_name, 
                            '" → "', NEW.product_name, '"; ');
    END IF;
    
    -- Track price changes
    IF OLD.price_usd != NEW.price_usd THEN
        SET changes = CONCAT(changes, 'price: $', OLD.price_usd, 
                            ' → $', NEW.price_usd, '; ');
    END IF;
    
    -- Track rating changes
    IF OLD.rating != NEW.rating THEN
        SET changes = CONCAT(changes, 'rating: ', IFNULL(OLD.rating, 0), 
                            ' → ', IFNULL(NEW.rating, 0), '; ');
    END IF;
    
    -- Insert log if changes detected
    IF changes != '' THEN
        INSERT INTO product_audit_logs (
            product_id, product_name, action_type, changed_fields
        )
        VALUES (NEW.product_id, NEW.product_name, 'UPDATE', changes);
    END IF;
END
View DELETE Trigger 
CREATE TRIGGER after_product_delete
AFTER DELETE ON products
FOR EACH ROW
BEGIN
    INSERT INTO product_audit_logs (
        product_id, product_name, action_type, changed_fields
    )
    VALUES (
        OLD.product_id, OLD.product_name, 'DELETE',
        CONCAT('Deleted product: ', OLD.product_name, ' ($', OLD.price_usd, ')')
    );
END
