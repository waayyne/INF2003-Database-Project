Nested Query 1: Products Above Average Rating

This query finds products whose rating is higher than the average rating of all products. 

SELECT p.product_id, p.product_name, b.brand_name, p.rating, p.price_usd
FROM products p, brands b
WHERE p.brand_id = b.brand_id
AND p.rating > (
    SELECT AVG(rating)
    FROM products
)
ORDER BY p.rating DESC
LIMIT 20;

Nested Query 2: Products From Brands With More Than 3 Products

This query finds products where the brand appears in a nested query that returns brands with more than 3 products. 

SELECT p.product_id, p.product_name, b.brand_name, p.rating, p.price_usd
FROM products p, brands b
WHERE p.brand_id = b.brand_id
AND p.brand_id IN (
    SELECT brand_id
    FROM products
    GROUP BY brand_id
    HAVING COUNT(*) > 3
)
ORDER BY b.brand_name, p.product_name
LIMIT 20;

Nested Query 3: Products Above Average Price

This query finds products that are more expensive than the average product price. 

SELECT p.product_id, p.product_name, b.brand_name, p.price_usd, p.rating
FROM products p, brands b
WHERE p.brand_id = b.brand_id
AND p.price_usd > (
    SELECT AVG(price_usd)
    FROM products
)
ORDER BY p.price_usd DESC

Nested Query 4: Brands Above Overall Average Rating

This query finds brands whose average product rating is higher than the overall average rating. 

SELECT b.brand_name, AVG(p.rating) AS avg_rating, COUNT(*) AS total_products
FROM products p, brands b
WHERE p.brand_id = b.brand_id
GROUP BY b.brand_name
HAVING AVG(p.rating) > (
    SELECT AVG(rating)
    FROM products
)
ORDER BY avg_rating DESC
LIMIT 20;

Nested Query 5: Highest Rated Products

This query finds products with the highest rating in the database. 

SELECT p.product_id, p.product_name, b.brand_name, p.rating, p.price_usd
FROM products p, brands b
WHERE p.brand_id = b.brand_id
AND p.rating = (
    SELECT MAX(rating)
    FROM products
)
ORDER BY p.product_name
LIMIT 20;

Nested Query 6: Products From Brands With Out-of-Stock Products

This query finds products from brands that have at least one out-of-stock product. 

SELECT p.product_id, p.product_name, b.brand_name, p.out_of_stock
FROM products p, brands b
WHERE p.brand_id = b.brand_id
AND p.brand_id IN (
    SELECT brand_id
    FROM products
    WHERE out_of_stock = 1
)
ORDER BY b.brand_name, p.product_name
LIMIT 20;
